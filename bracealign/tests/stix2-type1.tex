\section{\cs{underbrace} / \cs{overbrace}}

\subsection{Without \pkg{bracealign}}

\begin{dispExample}
  \[
    \overbrace{
      \overbrace{\medcontent}^{1}
      ~+~
      \overbrace{\smallcontent}^{2}
    }^{nested}
    ~=~
    \underbrace{\hugecontent}_{3}
    ~+~
    \overbrace{\bigcontent}^{4}
    ~+~
    \underbrace{\smallcontent}_{5}
  \]
\end{dispExample}
